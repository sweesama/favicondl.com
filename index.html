JavaScript (Fetch) 
async function getFavicon(domain, size = 128) {
  const url = `https://favicondl.com/api/favicon?domain=${domain}&size=${size}`;
  const response = await fetch(url);
  const blob = await response.blob();
  return URL.createObjectURL(blob);
}
Python (Requests) 
import requests

url = "https://favicondl.com/api/favicon?domain=stripe.com&size=256"
response = requests.get(url, allow_redirects=True)
with open("icon.png", "wb") as f:
    f.write(response.content)
